    <script>
        const inputField = document.getElementById('serial-input');
        inputField.addEventListener('input', function (e) {
            let target = e.target;
            let val = target.value.replace(/-/g, '').toUpperCase();
            let newVal = '';
            for (let i = 0; i < val.length; i++) {
                if (i > 0 && i % 4 === 0) {
                    newVal += '-';
                }
                newVal += val[i];
            }
            target.value = newVal;
        });

        document.addEventListener('click', () => {
            if (document.getElementById('lockdown-screen').style.display !== 'none') {
                inputField.focus();
            }
        });

        async function verifyCode(e) {
            e.preventDefault();
            const code = inputField.value.trim().toUpperCase();
            const errorMsg = document.getElementById('error-msg');
            const inputGroup = document.getElementById('input-group');

            try {
                const response = await fetch('/api/verify', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify({ code: code }),
                });

                const result = await response.json();

                if (result.success) {
                    document.body.style.backgroundColor = "#050505";
                    document.getElementById('lockdown-screen').style.display = "none";
                    const successScreen = document.getElementById('success-screen');
                    successScreen.style.display = "flex";
                } else {
                    errorMsg.style.display = "block";
                    inputGroup.classList.remove('shake');
                    void inputGroup.offsetWidth;
                    inputGroup.classList.add('shake');
                    inputField.value = "";
                    setTimeout(() => {
                        errorMsg.style.display = "none";
                    }, 3000);
                }
            } catch (error) {
                console.error('Error verifying code:', error);
            }
        }
    </script>
